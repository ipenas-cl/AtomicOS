// Constant-time cryptographic operations in Tempo
@constant_time
@wcet(1500)
function crypto_compare(a: ptr<int8>, b: ptr<int8>, len: int32) -> bool {
    let result: int32 = 0;
    
    // Compare all bytes regardless of match (no early exit)
    for let i = 0; i < len; i = i + 1 {
        result = result | (a[i] ^ b[i]);
    }
    
    // Constant-time conversion to boolean
    return ct_select(result == 0, true, false);
}

@wcet(500)
function secure_task() -> void {

        process_critical_data();
    }
}
