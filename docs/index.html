// Constant-time password comparison - immune to timing attacks!
@wcet(1000)              // Guaranteed 1000 cycles max
@constant_time           // No timing side-channels
fn compare_password(input: *[32]u8, stored: *[32]u8) -> bool {
    let diff: u32 = 0;
    
    // Every iteration takes exactly the same time
    for i in 0..32 {
        diff = diff | (input[i] ^ stored[i]);
    }
    
    return diff == 0;  // Secure comparison done right!
}
