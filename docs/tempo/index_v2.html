What is Tempo?

Tempo is a revolutionary systems programming language that guarantees predictable performance, memory safety, and security. It's designed for building real-time systems, operating systems, embedded software, and security-critical applications.

Quick Start

Get up and running with Tempo in minutes:

1. Install Tempo

# macOS (Homebrew)
brew install tempo

# From source
git clone https://github.com/tempo-lang/tempo
cd tempo
make install

2. Write Your First Program

// hello.tempo
module hello;

import io;

function main() -> int32 {
    io::println("Hello, Tempo!");
    return 0;
}

3. Compile and Run

tempo build hello.tempo
./hello

Real-Time Task

// Process sensor data with guaranteed timing
@wcet(1000)  // Must complete in 1000 cycles
function process_sensor_data(reading: float32) -> bool {
    if reading > THRESHOLD {
        activate_safety_system();
        return true;
    }
    return false;
}

Secure Password Comparison

// Constant-time comparison prevents timing attacks
@constant_time
function secure_compare(a: ptr, b: ptr, len: u32) -> bool {
    let diff = 0;
    for let i = 0; i < len; i++ {
        diff |= a[i] ^ b[i];
    }
    return diff == 0;
}

Zero-Copy Communication

// Efficient IPC with channels
channel sensor_data;

task sensor_reader {
    let msg = Message { 
        timestamp: get_time(),
        value: read_sensor()
    };
    sensor_data.send(msg);  // Zero-copy transfer
}

task data_processor {
    let msg = sensor_data.receive();
    process(msg);
}
