// Guaranteed to complete in 100 cycles
@wcet(100)
function process_packet(data: ptr<u8>, len: u32) -> bool {
    // Compiler ensures this meets timing constraint
    for let i = 0; i < len && i < 64; i++ {
        if !validate_byte(data[i]) {
            return false;
        }
    }
    return true;
}

// Prevent timing attacks automatically
@constant_time
function secure_compare(a: ptr<u8>, b: ptr<u8>, len: u32) -> bool {
    let diff = 0;
    for let i = 0; i < len; i++ {
        diff |= a[i] ^ b[i];
    }
    return diff == 0;
}
